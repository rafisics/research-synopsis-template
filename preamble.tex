  \ifentrytype{article}{%
    \clearfield{title}%
    \clearfield{subtitle}%
    
  }{}%
  \clearfield{note}%
  \clearfield{url}%
  \clearfield{urldate}%
  \clearfield{doi}%
  \clearfield{eprint}%
  \clearfield{issn}%
  \clearfield{isbn}%
  \clearfield{archiveprefix}%
  \clearfield{primaryclass}%
  \clearlist{language}%
  \clearfield{address}%

}

% Unified bibliography driver
\DeclareBibliographyDriver{article}{%
  \usebibmacro{author}%
  \setunit{\addcomma\space}\newblock
  \printfield{journaltitle}%
  \setunit{\addspace}\newblock
  \printtext[bold]{\printfield{volume}}%
  \setunit{\addcomma\space}\newblock
  \printfield{pages}%
  \setunit{\addspace}\newblock
  \printtext[parens]{\printfield{year}}%
  \newunit\newblock
  \usebibmacro{finentry}
}
\DeclareBibliographyDriver{book}{%
  \usebibmacro{author}%
  \setunit{\addcomma\space}\newblock
  \printfield{title}%
  \setunit{\addcomma\space}\newblock
  \printlist{publisher}%
  \setunit{\addspace}\newblock
  \printtext[parens]{\printfield{year}}%
  \newunit\newblock
  \usebibmacro{finentry}
}
